\subsection{Regning med funktioner}

\begin{align*}
f(x)=x^3+2x^2-8+10\\
g(x)=4x^3-5x^2+11\\
h(x)=16x^2-2x\\
i(x)=8x-1
\end{align*}

\textbf{Bestem:}
\begin{itemize}
	\item f(x)+g(x)
	\item g(x)-f(x)
	\item f(x)h(x)
	\item $\frac{h(x)}{i(x)}$ (Hint: faktorisere h(x) så du har $h(x) = \text{et faktor} \cdot i(x)$)
\end{itemize}
