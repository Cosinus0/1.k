\subsection*{Regnearternes Hierarki}
Parenteser (a+b) \\
Potenser og rødder $a^n$ $\sqrt[n]{a}$ \\
Multiplikation og division $a \cdot b$ $\frac{a}{b}$ $a:b$ \\
Addition og subtraktion $a+b$, $a-b$


\begin{itemize}
	\item $2 \cdot (3^2+2) -3$ \\
	\item 2 $\cdot (5^2+3\cdot 20-50)$
\end{itemize}

\subsection*{Ligninger}

\begin{itemize}
	\item $4x+2 = -(2x-1)$
	\item $x(1+x)=x^2+2$
\end{itemize}


\subsection*{Reduktion}

\begin{itemize}
	\item $-2b+3(b-2)$
	\item $x-2b+3(x-b)$
	\item $a+ab-2a+b-a(b+1)$
\end{itemize}

\subsection*{Kvadratsætninger}
$(a+b)^2=a^2+b^2+2ab$ \\
$(a-b)^2=a^2+b^2-2ab$ \\
$(a+b)(a-b)=a^-b^2$

\begin{itemize}
	\item $(x-5)(x+1)$
	\item $(x+a)^2$
	\item $(a-2b)^2$
\end{itemize}
